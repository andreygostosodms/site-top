  <script>
    function adicionar(valor) {
      document.getElementById("display").value += valor;
    }

    function limpar() {
      document.getElementById("display").value = "";
    }

    function calcular() {
      let display = document.getElementById("display");

      try {
        display.value = eval(display.value);
      } catch {
        display.value = "Erro";
      }
    }
  </script>
